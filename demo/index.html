在HTML中使用

<!-- 引入VeriGate组件 --> <script src="https://your-domain.netlify.app/verigate.js"></script> <!-- 创建容器 --> <div id="verigate-container"></div> <!-- 初始化组件 --> <script> VeriGate.render('#verigate-container', { siteKey: 'your-site-key', theme: 'light', onSuccess: function(token) { console.log('验证成功:', token); // 在这里处理验证成功的逻辑 } }); </script>

JavaScript API

// 渲染验证组件 const sessionId = VeriGate.render('#container', { siteKey: 'your-site-key', theme: 'light', // 'light' | 'dark' size: 'normal', // 'compact' | 'normal' | 'large' onSuccess: function(token) { // 验证成功回调 console.log('Token:', token); }, onError: function(error) { // 验证失败回调 console.error('Error:', error); } }); // 重置验证组件 VeriGate.reset(sessionId); // 客户端验证令牌 VeriGate.verify(token, siteKey).then(function(result) { console.log('验证结果:', result); });

服务端验证

// Node.js 示例 const response = await fetch('https://your-domain.netlify.app/.netlify/functions/verify', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ token: userToken, siteKey: 'your-site-key' }) }); const result = await response.json(); if (result.success) { console.log('验证成功', result.data); // 处理验证成功的逻辑 } else { console.error('验证失败', result.error); // 处理验证失败的逻辑 }
